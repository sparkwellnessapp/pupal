{

// בדיקה ראשונית: אורך זוגי - לא חובה ,זה יעילות

return false;

for (int i = 0; i < arr.Length; i++)

{

int target = -arr[i]; // המספר הנגדי שאנו מחפשים

bool found = false;

// לולאה פנימית לחיפוש המספר הנגדי במערך

{

if (arr[j] == target)

{

found = true; // מצאנו, זה יעצור את הלולאה מלהמשיך לחפש עבור איבר זה

// זימון הפעולה מסעיף א' לבדיקה האם הסדנאות דומות

if (this.workshops[i].IsSimilarWorkshop(ws))

{

// בדיקה האם סך המשתתפים אינו עולה על 40

if (this.workshops[i].GetPeople() + ws.GetPeople() <= 40)

{

// עדכון מספר המשתתפים בסדנה הקיימת

int updatedPeople = this.workshops[i].GetPeople() + ws.GetPeople();

this.workshops[i].SetPeople(updatedPeople);

joined = true; // סימון שנמצא פתרון, הלולאה תסתיים באופן טבעי

}

}

i++;

}

// אם הצלחנו לאחד את המשתתפים בסדנה קיימת, נחזיר true

if (joined)

return true;

// שלב 2: אם לא נמצאה סדנה מתאימה לאיחוד, ננסה להוסיף אותה כסדנה חדשה

if (this.countWorkshops < this.workshops.Length)        // אם יש מקום במערך

{

// הוספה למקום הראשון הפנוי

this.workshops[this.countWorkshops] = ws;

this.countWorkshops++; // עדכון מספר הסדנאות בפועל

return true;

}

// שלב 3: המערכת מלאה ולא ניתן לבצע איחוד או הוספה

return false;

}

// Q6 -start -  אלגוריתמיקה מתקדמת

public static int[] AnalyzeMechina(int[] studentsMechina)

{

int[] counters = new int[31]; // counters[i] = מספר תלמידים שנרשמו למכינה i

// מניחים שהשפה מאפסת את המערך מונים או עושים לולאה לאיפוס

// ספירת רישומים

for (int i = 0; i < studentsMechina.Length; i++)

{

int mechina = studentsMechina[i];

counters[mechina]++;

}

// שלב 2: ספירת המכינות שאליהן נרשם לפחות תלמיד אחד ומציאת המקסימום

int activeMechinasCount = 0;

int maxStudents = counters[1];

int maxMechinaNum = 1;

for (int i = 1; i <= 30; i++)

{

if (counters[i] > 0)

{

activeMechinasCount++;

}

// בדיקת המכינה הפופולרית ביותר

if (counters[i] > maxStudents)

{

maxStudents = counters[i];

maxMechinaNum = i;

}

}

// הדפסת המכינה שאליה נרשמו הכי הרבה תלמידים

Console.WriteLine("The mechina with the most students is: " + maxMechinaNum);

// שלב 3: הקצאת המערך המוחזר (כל מכינה פעילה תופסת 2 תאים: מספר מכינה וכמות)

int[] result = new int[activeMechinasCount * 2];

int resIndex = 0;

// שלב 4: מילוי המערך החדש בזוגות

for (int i = 1; i <= 30; i++)

{

if (counters[i] > 0)

{

result[resIndex] = i;              // תא ראשון: מספר המכינה

result[resIndex + 1] = counters[i]; // תא שני עוקב: מספר התלמידים

resIndex += 2;                     // קידום האינדקס ב-2 עבור הזוג הבא

}

}

return result;

}
