public static int[] Combine(int[] arr1, int[] arr2)

{

// חישוב האורך הכולל של המערך החדש

int totalLength = 0;

for (int i = 0; i < arr2.Length; i++)

{

totalLength += arr2[i];

}

// יצירת המערך החדש

int[] result = new int[totalLength];

int indexResult = 0;

// מילוי המערך החדש בהתאם לכללי השאלה

for (int i = 0; i < arr1.Length; i++)

{

for (int j = 1; j <= arr2[i]; j++)  //total iterations : arr2[i]

{

result[indexResult] = arr1[i];

indexResult++;

}

}

return result;

}
