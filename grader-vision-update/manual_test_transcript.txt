public class Employee // א.
{
    private int id;             
    private double salary;       
    private bool isSenior;   

    public Employee(int id, double salary, bool isSenior) // ב.
    {
        this.id = id;            
        this.salary = salary;    
        this.isSenior = isSenior;
    }


    public Employee(int id, double salary) // ג.
    {
        this.id = id;
        this.salary = salary;
        this.isSenior = false;   // לא ותיק
    }


    public double CalcMonthlySalary() // ד.
    {
    int add = 0;

    if (isSenior == false)
    {
        return salary; // מחזיר משכורת
    }
    else
    {
        add = salary * 0.3;   // תוספת
        salary += add;        // עדכון משכורת
        return salary;        // מחזיר משכורת
    }
}

    public class Department // ה.
    {
        private Employee[] arrEmployees; // עובדים

        public bool AddEmployee(Employee e)

            if (current < arrEmployees.Length)

                arrEmployees[current] = new Employee(e.GetId(), e.GetSalary(), e.GetIsSenior());
